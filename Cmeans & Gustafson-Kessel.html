def plot_points(points, fig=None, loc=(0, 1, 2), title='', method="Cmeans", m=10, n=4):
    if method == "Cmeans":
        algo = cm.FuzzyClustering(max_iter=m, n_clusters=n, method="Cmeans", debug=False)
    elif method == "GustafsonKessel":
        algo = cm.FuzzyClustering(max_iter=m, n_clusters=n, method="GustafsonKessel", debug=False)

    clusters, membership = algo.fit(points)
    index = membership.argmax(axis=1)
    if fig:
        fig.add_subplot(loc[0], loc[1], loc[2])
    colmap = {0: 'r', 1: 'g', 2: 'b', 3: 'y', 4: 'tab:pink'}
    for i in range(index.shape[0]):
        plt.scatter(points[i][0], points[i][1], c=colmap[index[i]])
    for i in range(clusters.shape[0]):
        plt.scatter(clusters[i][0], clusters[i][1], c='k')
    plt.title(title)
